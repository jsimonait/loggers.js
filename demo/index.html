Loggers.js

Logger best with firebug-lite

1. Simple logging

var l = loggers.create('Demo1');
l.log()('Log message');

2. Grouping 

var l = loggers.create('Demo2');
l.startLog('gKey1')('Groupe 1');
l.log()('Log message', 1);
l.log()('Log message', 2);
l.endLog('gKey1')('Groupe 1');

l.startLog('gKey2', true)('Groupe 2');
l.log()('Log message', 3);
l.log()('Log message', 4);
l.endLog('gKey2')('Groupe 2');

l.log()('Log message');

3. Filtering by key

var l = loggers.create('Demo3', {filters: ['Key1']});
l.log('Key1')('Log message', 1);
l.log('Key2')('Log message', 2);
l.log('Key1')('Log message', 3);
l.log('Key2')('Log message', 4);

4. Enable/Disable

var lEnabled = loggers.create('Demo4Enable', {enabled: true});
lEnabled.log()('Log message');

var lDisabled = loggers.create('Demo4Disable', {enabled: false});
lDisabled.log()('Log message');

5. Log info template

var l = loggers.create('Demo5', {
logInfoTemplate: '{name} ({time})',
});
l.log()('Log message');
